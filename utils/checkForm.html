# 表单验证

# 概述

在日常开发中，表单验证是比较常用的，前端库中也有一些比较流行的验证库，jq的Validator...等，然而因为目前以较少需要引入jq了，所以自己尝试封装一个简单的验证库。

# 实现

 const checkForm = {
  //判空
  isRequired: (value) => {
    if (!value || value == '') return false;
    return true;
  },

  isEmail: function (email) {
    if (!this.isRequired(email)) return false;
    let res = email.match(/^\w+([-+.]\w+)*@\w+([-.]\w+)*\.\w+([-.]\w+)*$/);
    if (res == null) return false;
    return true;
  },
  //11位的手机号
  isPhone: function (phone) {
    if (!this.isRequired(phone)) return false;
    let phoneReg = /(^1[3|4|5|7|8]\d{9}$)|(^09\d{8}$)/;
    if (!phoneReg.test(phone)) return false;
    return true;
  },
  //匹配密码，长度在6-16之间，只能输入数字跟字母
  isPwd: function (str) {
    if (!this.isRequired(str)) return false;
    var result = str.match(/^[A-Za-z0-9]{6,16}$/);
    if (result == null) return false;
    return true;
  },
  //传入一个表单data对象,后面的参数为数量不限的key字符串，会忽略其key验证
  checkObj: function (data,...ignores) {
    //在最后true的时候返回的验证成功过的Key数组,不包含忽略的key
    let keys = []
    //判断传进来的是否是一个对象
    if (!data || !(typeof data === 'object')) {
      return {isPass: false,msg:'is not a obj'}
    }
    for (const key in data) {
      if (data.hasOwnProperty(key)) {
        let isIgnore = false;
        ignores.forEach((item)=>{
          if (key === item) {
            isIgnore = true
          }
        })
        if (isIgnore) {
          continue
        }
        const value = data[key];
        if (!this.isRequired(value)) return {
          isPass: false,
          msg: `${key}为空`
        }    
        switch (key) {
          case 'phone':
            if (!this.isPhone(value)) return {
              isPass: false,
              msg: `${key}不符合要求`
            }
            break;
          case 'email':
            if (!this.isEmail(value)) return {
              isPass: false,
              msg: `${key}不符合要求`
            }
            break;
          case 'password':
            if (!this.isPwd(value)) return {
              isPass: false,
              msg: `${key}不符合要求`
            }
            break;
          default:
            break;
        }
        keys.push(key)
      }

    }
    return {
      isPass: true,
      keys
    }
  }
}

使用注意:

  1. 可通过模块引入checkForm对象
  2. 可单独验证某一项，比如checkForm.isPhone(val),返回true/false
  3. 可通过checkForm.checkObj(obj)传入form.data对象，默认对obj的属性判空（不含继承属性），对phone,email,password等key调用对应的方法。验证成功返回：{isPass: true,keys},验证失败返回: {isPass:false,msg:"err信息"}}
  4. 可通过在Obj后传入key字符串，会忽略该key的验证[验证成功后在keys中不会包含该key]
