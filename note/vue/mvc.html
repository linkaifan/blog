# mvc

传统的MVC主要分为三部分

  • 用户通过 View 传送指令到 Controller
  • Controller 完成业务逻辑后，要求 Model 改变状态
  • Model 将新的数据发送到 View，用户得到反馈

  class Model {
    constructor(value = '') {
      this._value = value
      this._listeners = []
    }
    set(value) {
      const self = this
      self._value = value
      // model中的值改变时，应通知注册过的回调函数
      // 按照Javascript事件处理的一般机制，我们异步地调用回调函数
      // 如果觉得setTimeout影响性能，也可以采用requestAnimationFrame
      setTimeout(() => {
        self._listeners.forEach((listener) => {
          listener.call(self, value)
        })
      });
    }
    watch(listener) {
      // 注册监听的回调函数
      this._listeners.push(listener);
    }
    bind(node) {
      // 将watch的逻辑和通用的回调函数放到这里
      this.watch(function (value) {
        node.innerHTML = value;
      });
    }
  }
  class Controller {
    constructor(callback) {
      let models = {}
      // 找到所有有bind属性的元素,处理成数组
      let views = Array.from(document.querySelectorAll('[data-bind]'))
      views.forEach((view) => {
        let modelName = view.getAttribute('data-bind');
        // 取出或新建该元素所绑定的model
        models[modelName] = models[modelName] || new Model();
        // 完成该元素和指定model的绑定
        models[modelName].bind(view);
      })
      // 调用controller的具体逻辑，将models传入，方便业务处理
      callback.call(this, models);
    }
  }
  new Controller(function (models) {   
    //在这里写业务逻辑改变model数据
    function setTime() {
      var date = new Date();
      models.hour.set(date.getHours());
      models.minute.set(date.getMinutes());
      models.second.set(date.getSeconds());
    }
    setTime();
    setInterval(setTime, 1000);
  });
