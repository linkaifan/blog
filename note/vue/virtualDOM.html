# 虚拟Dom

Vue 通过建立一个虚拟 DOM 对真实 DOM 发生的变化保持追踪。毕竟操作dom成本太高，1.0时候使用文档碎片。2.0开始引入vnode。

# 原理

  • Element: 返回vndoe，通过createElement方法调用
  • createElement(type,props,children)
  • render(eleObj): 将createElement返回的vnode渲染成真实的dom
  • renderDom(el,target): 将render返回的dom挂载到target
  class Element{
    constructor(type,props,children){
      this.type = type
      this.props = props
      this.children = children 
    }
  }
  function createElement(type,props,children) {
    return new Element(type,props,children)
  }
  function setAttr(node,key,value) {
    switch (key) {
      case 'value'://node是input或者textarea
        if (node.tagName.toUpperCase === 'INPUT'||
        node.tagName.toUpperCase === 'TEXTAREA') {
          node.value = value
        }else{
          node.setAttribute('value',value)
        }
        break;
      case 'style':
        node.style.cssText = value
        break
      default:
        node.setAttribute(key,value)
        break;
    }
  }
  function render(eleObj) {
    let el = document.createElement(eleObj.type)
    for (const key in eleObj.props) {
      //设置属性的方法
      setAttr(el,key,eleObj.props[key])
    }
    eleObj.children.forEach(child => {
      child = (child instanceof Element)?
      render(child):document.createTextNode(child)
      el.appendChild(child)
    });
    return el
  }
  //将render返回的el挂载到target上
  function renderDom(el,target) {
    target.appendChild(el)
  }
  let vnode = createElement("ul",{class:"list"},[
    createElement("li",{class:"item"},['a']),
    createElement("li",{class:"item"},['b']),
    createElement("li",{class:"item"},['c'])
  ])
  let el = render(vnode)
  renderDom(el,document.getElementById("app"))
  console.log(el);

# Dom Diff

比较两个虚拟dom区别=>两个对象的区别，用于当vnode发生改变，通过创建补丁patch对象，描述改变的内容，用最小的开销去更新dom

# 原理

diff生成patches，内部原理是：walk递归，全局维护Index【每次diffChild时加1，传递给walk的时候因为闭包私有化】，patches记录Index对应的vnode变化【如果有】，这里只考虑四种情况：属性变化ATTRS，文本变化TEXT，节点删除REMOVE,节点替换REPLACE。【当节点type不变时，需要diffChild进行深度优先遍历】

  const ATTRS = 'ATTRS'
  const TEXT = 'TEXT'
  const REMOVE = 'REMOVE'
  const REPLACE = 'REPLACE'
  let Index = 0
  //差异算法
  function diff(oldTree,newTree) {
    let patches = {}
    //递归，比较后的结果放到补丁包中
    walk(oldTree,newTree,0,patches)
    return patches
  }
  function walk(oldNode,newNode,index,patches) {
    //每个元素都有一个补丁对象
    let currentPatch = {}
    if (!newNode) {
      //节点删除
      currentPatch = {type:REMOVE,index}
    }else if (isString(oldNode) && isString(newNode)) {
      //文本节点
      if (oldNode !== newNode ) {
        currentPatch = {type:TEXT,text:newNode}
      }
    }else if (oldNode.type === newNode.type) {
      //比较属性是否有更改        
      let attrs = diffAttr(oldNode.props,newNode.props)      
      if (Object.keys(attrs).length) {
        currentPatch = {type:ATTRS,attrs}
      }
      //存在子节点，递归
      diffChildren(oldNode.children,newNode.children,patches)
    }else{
      //节点被替换
      currentPatch = {type:REPLACE,newNode} 
    }
    currentPatch.type && (patches[index] = currentPatch)             
  }
  function diffAttr(oldAttrs,newAttrs) {
    let patch = {}
    for (const key in oldAttrs) {
      if (oldAttrs[key] !== newAttrs[key]) {
        patch[key] = newAttrs[key]      //有可能是undefined
      }
    }
    //判断是否有新增属性
    for (const key in newAttrs) {
      if (!oldAttrs.hasOwnProperty(key)) {
        patch[key] = newAttrs[key]      
      }
    }           
    return patch
  }
  function diffChildren(oldChildren,newChildren,patches) {
    oldChildren.forEach((child,idx)=>{
      walk(child,newChildren[idx],++Index,patches)           
    })
  }

# patch

patch传入node【已经render】,和diff生成的patches，全局维护patchIndex，patch.type去判断diff对应的4种情况。

  let patchIndex = 0
  let allPatches
  //打补丁
  function patch(node,patches) {
    allPatches = patches
    patchWalk(node)
  }
  function patchWalk(node) {
    let currentPatch = allPatches[patchIndex++]  
    let childNodes = node.childNodes
    childNodes.forEach(child=>patchWalk(child))
    if (currentPatch) {
      doPatch(node,currentPatch)
    }
  }
  function doPatch(node,patch) {  
    switch (patch.type) {
      case ATTRS:
        for (const key in patch.attrs) {
          let value = patch.attrs[key]
          if (value) {
            setAttr(node,key,value)
          }else{
            node.removeAttribute(key)
          }
        }
        break;
      case TEXT:        
        node.textContent = patch.text
      break;
      case REPLACE:
        let newNode = (patch.newNode instanceof Element)?render(patch.newNode):
          document.createTextNode(newNode)
        node.parentNode.replaceChild(newNode,node)
      break;
      case REMOVE:
        node.parentNode.removeChild(node)
      break;        
      default:
        break;
    }
  }
  let vnode1 = createElement("ul",{class:"list"},[
    createElement("li",{class:"item"},['a']),
    createElement("li",{class:"item"},['b']),
    createElement("li",{class:"item"},['c'])
  ])
  let vnode2 = createElement("ul",{class:"list-group",id:"a"},[
    createElement("li",{class:"item"},['1']),
    createElement("li",{class:"item"},['b']),
    // createElement("div",{class:"item"},['3'])
  ]) 

  let el = render(vnode1)
  renderDom(el,document.getElementById("app"))
  let patches = diff(vnode1,vnode2)
  //打补丁，重新更新视图
  patch(el,patches)
  console.log(patches);
